```C++
#include <cstdio>
#include <cctype>
const int MOD = 19260817;
inline int getint()
{
    int res = 0, ch = getchar();
    while(!isdigit(ch) and ch != EOF)
        ch = getchar();
    while(isdigit(ch))
    {
        res = (res << 3) + (res << 1) + (ch - '0');
        res %= MOD;
        ch = getchar();
    }
    return res;
}

int x, y;
void exgcd(int a, int b)
{
    if(b == 0)
    {
        x = 1;
        y = 0;
        return;
    }
    exgcd(b, a % b);
    int Last_x = x;
    x = y;
    y = Last_x - a / b * y;
}

int main()
{
	int a, b;
    a = getint();
    b = getint();
    
    if(b == 0)
    {
        puts("Angry!");
        return 0;
    }
    exgcd(b, MOD);
    x = (x % MOD + MOD) % MOD;
    printf("%lld\n", a * (long long)(x) % MOD);
    return 0;
}
```

3.线性筛

```c++
#include <iostream>
#include <vector>
#include <cmath>

#define int long long

using namespace std;

const int maxn = 1000000;
int prime[maxn], cnt;
bool vis[maxn];
bool ans[maxn];

void Euler() {
    for (int i = 2; i <= 50000; ++i) {
        if (!vis[i]) {
            prime[++cnt] = i;
            for (int j = i + i; j <= 50000; j += i) {
                vis[j] = true;
            }
        }
    }
}

int _max(int a, int b) {
    return a > b ? a : b;
}

signed main() {
    Euler();
    int l, r;
    cin >> l >> r;
    int tot = 0;

    // 初始化 ans 数组
    fill(ans, ans + r - l + 1, true);

    for (int i = 1; i <= cnt; ++i) {
        int start = _max(2, (l - 1) / prime[i] + 1) * prime[i];
        for (int j = start; j <= r; j += prime[i]) {
            if (j >= l) {
                ans[j - l] = false;
            }
        }
    }

    // 统计质数数量
    for (int i = 0; i <= r - l; ++i) {
        if (ans[i]) {
            tot++;
        }
    }

    // 特殊情况处理：如果 l 是 1，则需要减去 1，因为它不是质数
    if (l == 1) {
        tot--;
    }

    cout << tot << endl;
    return 0;
}
```

```c++
#include <cstdio>

const int maxn = 1e6 + 5;
int num[maxn], a[maxn], sum[maxn];
int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	memset(num, 0, sizeof(num));
	memset(sum, 0, sizeof(sum));
	for(int i = 0; i < n; ++i){
		scanf("%d", &a[i]);
		if(a[i] <= m){
			++num[a[i]];
		}
	}
	for(int i = 1; i <= m; ++i){
		if(num[i]){
			for(int j = i; j <= m; j += i){
				sum[j] += num[i];
			}
		}
	}
	int max_sum = 0, max_val = 0;
	for(int i = 1; i <= m; ++i){
		if(sum[i] && max_sum < sum[i]){
			max_val = i;
			max_sum = sum[i];
		}
	}
	if(max_sum == 0){
		printf("1 0\n\n");
	}else{
		int cas = 0;
		printf("%d %d\n", max_val, max_sum);
		for(int i = 0; i < n; ++i){
			if(max_val % a[i] == 0){
				if(cas){
					printf(" ");
				}
				printf("%d", i + 1);
				++cas;
			}
		}
		puts("");
	}
```

```c++
#include <iostream>
#include <cstring>
#include <algorithm>
#include<unordered_map>
#define int long long

const int N=1000010;
int prim[N];
int cot;
unordered_map<int,int> st;
int a[N];
void init()
{
    for(int i=2;i<=N;i++)
    {
        if(!st[i])
           prim[cot++]=i;
        for(int j=0;prim[j]<=N/i;j++)
        {
            st[i*prim[j]]=1;
            if(i%prim[j]==0)
              break;
        }
    }
}
signed main()
{
    init();
    int t;
    cin>>t;
    int kk=0;

    {

           kk=n;

             cnt++;
        }

          cout<<2<<'\n';
        else if(cnt>=2)
          cout<<-1<<'\n';
        else if(cnt==1&&st[mi])

            int flo=1;
```
